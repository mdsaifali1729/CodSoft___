#include <iostream>
#include <fstream>
#include <string>
using namespace std;
int main()
{   string NameOfFile;
    cout << "Enter name of the text file:\n";
    cin >> NameOfFile;
    ifstream file(NameOfFile);
    if (!file)
        {
            cout << "File cdoesn't exist!\n";
            return 1;
        }
    int WordCount = 0;
    string word;
    while (file >> word) WordCount++;
    cout << "Total number of words in text file: " << WordCount << endl;
    return 0;
}
